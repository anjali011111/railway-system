Trigger 1 — Check Seat Before Booking
DELIMITER $$
CREATE TRIGGER before_ticket_insert
BEFORE INSERT ON Tickets FOR EACH ROW
BEGIN
  DECLARE seat_taken INT;
  SELECT COUNT(*) INTO seat_taken
  FROM Tickets
  WHERE train_id=NEW.train_id AND seat_number=NEW.seat_number AND status='Booked';
  IF seat_taken > 0 THEN
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT='Seat already booked';
  END IF;
END$$
DELIMITER ;
Trigger 2 — Log After Booking
DELIMITER $$
CREATE TRIGGER after_ticket_booking
AFTER INSERT ON Tickets FOR EACH ROW
BEGIN
  INSERT INTO BookingLog(ticket_id,passenger_id,action,action_time)
  VALUES(NEW.ticket_id,NEW.passenger_id,'BOOKING_CONFIRMED',NOW());
END$$
DELIMITER ;
Trigger 3 — Log Cancellation
DELIMITER $$
CREATE TRIGGER after_ticket_cancel
AFTER UPDATE ON Tickets FOR EACH ROW
BEGIN
  IF NEW.status='Cancelled' AND OLD.status='Booked' THEN
    INSERT INTO CancellationLog(ticket_id,passenger_id,train_id,cancelled_at)
    VALUES(OLD.ticket_id,OLD.passenger_id,OLD.train_id,NOW());
  END IF;
END$$
DELIMITER ;
Procedure 4 — Book a Ticket
DELIMITER $$
CREATE PROCEDURE BookTicket(
  IN p_passenger_id INT, IN p_train_id INT,
  IN p_seat_number VARCHAR(10), IN p_booking_date DATE
)
BEGIN
  INSERT INTO Tickets(passenger_id,train_id,booking_date,seat_number,status)
  VALUES(p_passenger_id,p_train_id,p_booking_date,p_seat_number,'Booked');
  SELECT LAST_INSERT_ID() AS new_ticket_id;
END$$
DELIMITER ;
-- Usage: CALL BookTicket(1, 1, 'A1', '2025-06-01');
Procedure 5 — Passenger History
DELIMITER $$
CREATE PROCEDURE GetPassengerHistory(IN p_passenger_id INT)
BEGIN
  SELECT t.ticket_id, tr.name AS train_name,
         tr.source, tr.destination,
         t.booking_date, t.seat_number, t.status
  FROM Tickets t
  JOIN Trains tr ON t.train_id=tr.train_id
  WHERE t.passenger_id=p_passenger_id
  ORDER BY t.booking_date DESC;
END$$
DELIMITER ;
-- Usage: CALL GetPassengerHistory(1);
Procedure 6 — Cancel a Ticket
DELIMITER $$
CREATE PROCEDURE CancelTicket(IN p_ticket_id INT)
BEGIN
  UPDATE Tickets SET status='Cancelled'
  WHERE ticket_id=p_ticket_id AND status='Booked';
  SELECT ROW_COUNT() AS rows_affected;
END$$
DELIMITER ;
-- Usage: CALL CancelTicket(2);
View 7 — Active Bookings
CREATE VIEW ActiveBookings AS
SELECT t.ticket_id, p.name AS passenger,
       tr.name AS train, tr.source, tr.destination,
       t.seat_number, t.booking_date
FROM Tickets t
JOIN Passengers p  ON t.passenger_id=p.passenger_id
JOIN Trains     tr ON t.train_id=tr.train_id
WHERE t.status='Booked';
View 8 — Train Occupancy
CREATE VIEW TrainOccupancy AS
SELECT tr.name, tr.source, tr.destination,
       COUNT(t.ticket_id) AS booked_seats
FROM Trains tr
LEFT JOIN Tickets t ON tr.train_id=t.train_id
  AND t.status='Booked'
GROUP BY tr.train_id;
View 9 — Passenger Summary
CREATE VIEW PassengerSummary AS
SELECT p.passenger_id, p.name, p.email,
       COUNT(t.ticket_id) AS total_bookings,
       SUM(CASE WHEN t.status='Booked'    THEN 1 ELSE 0 END) AS active,
       SUM(CASE WHEN t.status='Cancelled' THEN 1 ELSE 0 END) AS cancelled
FROM Passengers p
LEFT JOIN Tickets t ON p.passenger_id=t.passenger_id
GROUP BY p.passenger_id;
Nested 10 — Trains with Above Avg Bookings
SELECT tr.name, COUNT(t.ticket_id) AS bookings
FROM Trains tr
JOIN Tickets t ON tr.train_id=t.train_id
GROUP BY tr.train_id
HAVING COUNT(t.ticket_id) > (
  SELECT AVG(cnt) FROM(
    SELECT COUNT(*) AS cnt FROM Tickets
    GROUP BY train_id
  ) sub
);
Nested 11 — Top Revenue Train
SELECT tr.name, sub.total * 500 AS revenue
FROM Trains tr
JOIN (
  SELECT train_id, COUNT(*) AS total
  FROM Tickets WHERE status='Booked'
  GROUP BY train_id
) sub ON tr.train_id=sub.train_id
ORDER BY sub.total DESC LIMIT 1;
Nested 12 — Cancelled Last Month
SELECT t.ticket_id, p.name, tr.name AS train, t.seat_number
FROM  Tickets    t
JOIN  Passengers p  ON t.passenger_id=p.passenger_id
JOIN  Trains     tr ON t.train_id=tr.train_id
WHERE t.status='Cancelled'
  AND t.booking_date >= (
    SELECT DATE_SUB(CURDATE(), INTERVAL 1 MONTH)
  );
Join 13 — Full Ticket Details
SELECT t.ticket_id,
       p.name  AS passenger, p.email, p.phone,
       tr.name AS train, tr.source, tr.destination,
       t.seat_number, t.booking_date, t.status
FROM  Tickets    t
JOIN  Passengers p  ON t.passenger_id=p.passenger_id
JOIN  Trains     tr ON t.train_id=tr.train_id
ORDER BY t.ticket_id;
Join 14 — Trains from a Station
SELECT tr.name, tr.destination, tr.departure_time,
       COUNT(t.ticket_id) AS tickets_booked
FROM      Trains  tr
LEFT JOIN Tickets t ON tr.train_id=t.train_id AND t.status='Booked'
WHERE tr.source='Delhi'
GROUP BY tr.train_id;
Join 15 — Passenger Booked Tickets
SELECT t.ticket_id, tr.name, tr.source, tr.destination,
       tr.departure_time, t.seat_number, t.booking_date
FROM  Tickets    t
JOIN  Passengers p  ON t.passenger_id=p.passenger_id
JOIN  Trains     tr ON t.train_id=tr.train_id
WHERE p.passenger_id=1 AND t.status='Booked'
ORDER BY t.booking_date DESC;

DDL — Create Tables
CREATE TABLE Trains(
  train_id       INT AUTO_INCREMENT PRIMARY KEY,
  name           VARCHAR(100) NOT NULL,
  source         VARCHAR(100) NOT NULL,
  destination    VARCHAR(100) NOT NULL,
  departure_time DATETIME NOT NULL,
  arrival_time   DATETIME NOT NULL
);
CREATE TABLE Passengers(
  passenger_id INT AUTO_INCREMENT PRIMARY KEY,
  name  VARCHAR(100) NOT NULL,
  email VARCHAR(100) UNIQUE NOT NULL,
  phone VARCHAR(15)
);
CREATE TABLE Tickets(
  ticket_id    INT AUTO_INCREMENT PRIMARY KEY,
  passenger_id INT NOT NULL,
  train_id     INT NOT NULL,
  booking_date DATE NOT NULL,
  seat_number  VARCHAR(10) NOT NULL,
  status       VARCHAR(20) DEFAULT 'Booked',
  FOREIGN KEY(passenger_id) REFERENCES Passengers(passenger_id),
  FOREIGN KEY(train_id)     REFERENCES Trains(train_id)
);
